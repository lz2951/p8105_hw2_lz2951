## Clean the data in pols-month.csv

``` r
pols_month = 
  read_csv(file = "./data/pols-month.csv") |> 
  janitor::clean_names() |>
  separate(mon, into = c("year", "month", "day"), sep = "-", convert = TRUE) |>
  mutate(
    month = case_match(
      month,
      1 ~ "January",
      2 ~ "February",
      3 ~ "March",
      4 ~ "April",
      5 ~ "May",
      6 ~ "June",
      7 ~ "July",
      8 ~ "August",
      9 ~ "September",
      10 ~ "October",
      11 ~ "November",
      12 ~ "December"
    )
  ) |>
  pivot_longer(
    c("prez_dem", "prez_gop"),
    names_to = "president",
    names_prefix = "prez_",
    values_to = "party"
  ) |>
  filter(party == 1) |>
  select(-party, -day)
```
